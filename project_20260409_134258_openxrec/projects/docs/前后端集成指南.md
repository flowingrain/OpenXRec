## 前端调用示例

### 1. 序列推荐

```typescript
// src/services/api.ts

export interface Behavior {
  itemId: string;
  itemTitle: string;
  itemType?: string;
  action: string;
  timestamp: number;
  rating?: number;
  duration?: number;
  context?: Record<string, any>;
}

export interface CandidateItem {
  id: string;
  title: string;
  type?: string;
}

export interface SequenceRecommendationRequest {
  userId: string;
  behaviors: Behavior[];
  candidateItems: CandidateItem[];
  limit?: number;
  useBaseline?: boolean;
}

export interface Recommendation {
  item: Record<string, any>;
  score: number;
  reasoning: string;
  strategy: string;
  patternMatch?: Record<string, any>;
  confidence: number;
}

export interface SequenceAnalysis {
  userId: string;
  totalBehaviors: number;
  uniqueItems: number;
  patterns: Array<{
    pattern: string[];
    patternType: string;
    frequency: number;
    confidence: number;
    lastSeen: number;
    nextPredictions: string[];
  }>;
  insights: string[];
  recommendations: Recommendation[];
}

// API调用函数
export async function sequentialRecommendation(
  request: SequenceRecommendationRequest
): Promise<{ success: boolean; data: SequenceAnalysis }> {
  const response = await fetch('http://localhost:5000/api/recommendation/sequential', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify(request),
  });

  if (!response.ok) {
    throw new Error(`API调用失败: ${response.statusText}`);
  }

  return response.json();
}
```

### 2. 动态配置 - 收集反馈

```typescript
export interface UserFeedback {
  userId: string;
  itemId: string;
  sessionId: string;
  timestamp: number;
  action: string;
  recommendationId?: string;
  duration?: number;
  rating?: number;
  context?: Record<string, any>;
}

export async function collectFeedback(
  userId: string,
  feedback: UserFeedback
): Promise<{ success: boolean; message: string; status: { pendingCount: number; isAnalyzing: boolean } }> {
  const response = await fetch('http://localhost:5000/api/recommendation/dynamic-config', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      userId,
      action: 'collect',
      feedback,
    }),
  });

  return response.json();
}
```

## React组件示例

### 序列推荐演示组件

```typescript
'use client';

import { useState } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { sequentialRecommendation } from '@/services/api';

export function SequentialRecommendationDemo() {
  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<any>(null);

  const exampleBehaviors = [
    {
      itemId: 'item_1',
      itemTitle: 'Python入门教程',
      itemType: 'course',
      action: 'view',
      timestamp: Date.now() - 86400000 * 7
    },
    {
      itemId: 'item_2',
      itemTitle: 'Python实战项目',
      itemType: 'course',
      action: 'like',
      timestamp: Date.now() - 86400000 * 6
    }
  ];

  const exampleCandidates = [
    { id: 'item_4', title: 'Django框架教程', type: 'course' },
    { id: 'item_5', title: 'JavaScript入门', type: 'course' }
  ];

  const handleRecommend = async () => {
    setLoading(true);
    try {
      const response = await sequentialRecommendation({
        userId: `user_${Date.now()}`,
        behaviors: exampleBehaviors,
        candidateItems: exampleCandidates,
        limit: 5
      });

      if (response.success) {
        setResult(response.data);
      }
    } catch (error) {
      console.error('推荐失败:', error);
      alert('推荐失败，请查看控制台');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div>
      <Card>
        <CardHeader>
          <CardTitle>序列推荐（Python后端）</CardTitle>
        </CardHeader>
        <CardContent>
          <Button onClick={handleRecommend} disabled={loading}>
            {loading ? '分析中...' : '生成推荐'}
          </Button>
        </CardContent>
      </Card>

      {result && (
        <Card>
          <CardHeader>
            <CardTitle>推荐结果</CardTitle>
          </CardHeader>
          <CardContent>
            {result.recommendations.map((rec: any, index: number) => (
              <div key={index} className="p-4 border rounded-lg mb-2">
                <h3 className="font-semibold">{rec.item.title}</h3>
                <p className="text-sm text-muted-foreground">{rec.reasoning}</p>
                <Badge>{rec.strategy}</Badge>
              </div>
            ))}
          </CardContent>
        </Card>
      )}
    </div>
  );
}
```

### 动态配置演示组件

```typescript
'use client';

import { useState } from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { collectFeedback, triggerAnalysis, getDynamicConfig } from '@/services/api';

export function DynamicConfigDemo() {
  const [userId] = useState(`user_${Date.now()}`);
  const [config, setConfig] = useState<any>(null);
  const [bufferStatus, setBufferStatus] = useState({ pendingCount: 0, isAnalyzing: false });

  const loadConfig = async () => {
    try {
      const response = await getDynamicConfig(userId);
      if (response.success) {
        setConfig(response.currentConfig);
        setBufferStatus(response.bufferStatus);
      }
    } catch (error) {
      console.error('加载配置失败:', error);
    }
  };

  const simulateFeedback = async () => {
    try {
      await collectFeedback(userId, {
        userId,
        itemId: `item_${Date.now()}`,
        sessionId: `session_${Date.now()}`,
        timestamp: Date.now(),
        action: 'click',
        context: {
          scenario: 'product_recommendation',
          strategy: 'sequential'
        }
      });

      // 更新状态
      setBufferStatus(prev => ({
        ...prev,
        pendingCount: prev.pendingCount + 1
      }));

      // 如果达到10条，自动触发分析
      if (bufferStatus.pendingCount + 1 >= 10) {
        await handleAnalyze();
      }
    } catch (error) {
      console.error('收集反馈失败:', error);
    }
  };

  const handleAnalyze = async () => {
    try {
      const response = await triggerAnalysis(userId, 'product_recommendation');
      if (response.success) {
        alert(response.reasoning);
        await loadConfig();
        setBufferStatus({ pendingCount: 0, isAnalyzing: false });
      }
    } catch (error) {
      console.error('分析失败:', error);
    }
  };

  return (
    <div>
      <Card>
        <CardHeader>
          <CardTitle>动态配置（Python后端）</CardTitle>
          <CardDescription>
            待分析：{bufferStatus.pendingCount}/10
          </CardDescription>
        </CardHeader>
        <CardContent>
          <Button onClick={simulateFeedback}>模拟反馈</Button>
          <Button onClick={handleAnalyze}>触发分析</Button>
          <Button onClick={loadConfig}>加载配置</Button>
        </CardContent>
      </Card>

      {config && (
        <Card>
          <CardHeader>
            <CardTitle>当前配置</CardTitle>
          </CardHeader>
          <CardContent>
            <pre>{JSON.stringify(config, null, 2)}</pre>
          </CardContent>
        </Card>
      )}
    </div>
  );
}
```

## 部署建议

### 开发环境

```bash
# 1. 启动Python后端
cd /workspace/projects
python app/main.py

# 2. 启动Next.js前端（另一个终端）
cd /workspace/projects
pnpm dev
```

### 生产环境

#### 方案1：分离部署

```bash
# 后端
gunicorn app.main:app -w 4 -k uvicorn.workers.UvicornWorker -b 0.0.0.0:5000

# 前端
pnpm build
pnpm start
```

#### 方案2：Docker Compose

```yaml
# docker-compose.yml
version: '3.8'

services:
  backend:
    build: .
    ports:
      - "5000:5000"
    environment:
      - COZE_API_KEY=${COZE_API_KEY}
      - COZE_BOT_ID=${COZE_BOT_ID}

  frontend:
    build: ./frontend
    ports:
      - "3000:3000"
    environment:
      - NEXT_PUBLIC_API_URL=http://backend:5000
```

## 跨域处理

Python后端已配置CORS，支持所有来源：

```python
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)
```

```python
import os

app.add_middleware(

    allow_credentials=True,
```
