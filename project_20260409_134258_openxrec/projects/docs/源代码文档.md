  state: WorkflowState,

  const startTime = Date.now();
  console.log('[IntentParser] Analyzing query:', state.query);
  
  try {

用户查询：${state.query}
领域：${state.domain || '通用'}

请分析：
1. 意图类型（趋势分析、事件解读、风险评估、政策影响等）
2. 任务复杂度（low/medium/high）
3. 需要的智能体组合
4. 执行模式建议

输出JSON格式：
{
  "intentType": "trend_analysis|event_interpretation|risk_assessment|policy_impact",
  "complexity": "low|medium|high",
  "requiredAgents": ["agent1", "agent2", ...],
  "executionMode": "sequential|parallel|conditional",
  "reasoning": "分析推理过程"
}`;

    const response = await llm.invoke([{
      role: 'user',
      content: prompt
    }], {
      model: 'doubao-seed-2-0-pro-260215',
      temperature: 0.1
    });

    const text = response.content || '';
    const jsonMatch = text.match(/\{[\s\S]*\}/);
    
    if (jsonMatch) {
      const result = JSON.parse(jsonMatch[0]);
      
      // 确保核心智能体始终包含
      const coreAgents = ['scout_cluster', 'event_extractor', 'report_generator'];
      const requiredAgents = [...new Set([...coreAgents, ...result.requiredAgents])];
      
      const schedule: AgentSchedule = {
        requiredAgents,
        executionMode: result.executionMode || 'sequential',
        estimatedComplexity: result.complexity || 'medium',
        reasoning: result.reasoning
      };
      
      return {
        agentSchedule: schedule,
        currentAgent: 'intent_parser',
        status: 'intent_parsing'
      };
    }
    
    // 默认调度
    return {
      agentSchedule: {
        requiredAgents: ['scout_cluster', 'event_extractor', 'timeline_analyst', 
                         'causal_analyst', 'scenario_simulator', 'report_generator'],
        executionMode: 'sequential',
        estimatedComplexity: 'medium',
        reasoning: '默认分析流程'

      currentAgent: 'intent_parser',
      status: 'intent_parsing'

      error: `意图解析失败: ${error}`,

/**

 * 多源信息采集、实时数据抓取
 */
export async function scoutClusterNode(
  state: WorkflowState,
  config?: RunnableConfig
): Promise<Partial<WorkflowState>> {
  console.log('[ScoutCluster] Gathering information for:', state.query);
  
  try {
    const searchResults = await webSearchTool(state.query, {
      maxResults: 15,
      includeNews: true,
      includeAcademic: true
    });
    
    // 评估信息质量
    const evaluatedResults = searchResults.map(result => ({
      ...result,
      credibility: calculateCredibility(result)
    }));
    
    return {
      searchResults: evaluatedResults,
      currentAgent: 'scout_cluster',
      status: 'information_gathering'
    };
    
  } catch (error) {
    console.error('[ScoutCluster] Error:', error);
    return {
      error: `信息采集失败: ${error}`,
      searchResults: []
    };
  }
}

/**
 * 计算信息可信度
 */
function calculateCredibility(result: { source: string; url: string }): number {
  let score = 0.5;
  
  // 来源可信度加成
  const trustedSources = ['gov.cn', 'edu.cn', 'org.cn', 'reuters.com', 'bloomberg.com'];
  if (trustedSources.some(s => result.url.includes(s))) {
    score += 0.3;
  }
  
  // 官方来源加成
  if (result.source.includes('官方') || result.source.includes('政府')) {
    score += 0.2;
  }
  
  return Math.min(score, 1.0);
}

/**
 * 时序分析师节点
 * 事件时间线构建、发展脉络梳理
 */
export async function timelineAnalystNode(
  state: WorkflowState,
  config?: RunnableConfig
): Promise<Partial<WorkflowState>> {
  console.log('[TimelineAnalyst] Building timeline');
  
  if (!state.events || state.events.length === 0) {
    return { timeline: [] };
  }
  
  try {
    // 按时间排序事件
    const sortedEvents = [...state.events].sort((a, b) => 
      new Date(a.timestamp).getTime() - new Date(b.timestamp).getTime()
    );
    
    // 识别趋势变化
    const timelineWithTrend = sortedEvents.map((event, index, arr) => {
      let trendChange: 'up' | 'down' | 'stable' = 'stable';
      
      if (index > 0) {
        const prevEvent = arr[index - 1];
        if (event.heatIndex && prevEvent.heatIndex) {
          trendChange = event.heatIndex > prevEvent.heatIndex ? 'up' : 
                        event.heatIndex < prevEvent.heatIndex ? 'down' : 'stable';
        }
      }
      
      return {
        ...event,
        trendChange,
        weight: event.heatIndex || 0.5
      };
    });
    
    return {
      timeline: timelineWithTrend,
      currentAgent: 'timeline_analyst',
      status: 'analyzing'
    };
    
  } catch (error) {
    console.error('[TimelineAnalyst] Error:', error);
    return { timeline: state.events };
  }
}

  if (!state.finalReport || state.finalReport.length < 500) {

  if (!state.scenarios || state.scenarios.length < 2) {

  if (highRiskFactors && highRiskFactors.length > 3) {
    issues.push('存在多个高置信度负面因素');
    riskLevel = 'high';
  }
  
  return {
    qualityCheckResult: {
      passed: issues.length === 0,
      issues,
      riskLevel
    },

/**
 * 根据调度结果路由到下一个节点
 */
function routeBySchedule(state: WorkflowState): string {
  const schedule = state.agentSchedule;
  
  if (!schedule || !schedule.requiredAgents.length) {
    return 'report_generator';
  }
  
  // 获取当前执行的智能体索引
  const currentAgent = state.currentAgent;
  const currentIndex = schedule.requiredAgents.indexOf(currentAgent);
  
  if (currentIndex === -1 || currentIndex >= schedule.requiredAgents.length - 1) {
    return 'report_generator';
  }
  
  return schedule.requiredAgents[currentIndex + 1];
}

/**
 * 感知体层完成后的路由

function routeAfterPerception(state: WorkflowState): string {
  const schedule = state.agentSchedule;

    return 'timeline_analyst';
  }
  
  return 'report_generator';
}

/**
 * 创建工作流图
 */
export function createWorkflowGraph() {
  // 创建状态图
  const workflow = new StateGraph(WorkflowStateAnnotation);
  
  // ==================== 添加节点 ====================
  
  // 调度层
  workflow.addNode('intent_parser', agentNodes.intent_parser);
  
  // 感知体层
  workflow.addNode('scout_cluster', agentNodes.scout_cluster);
  workflow.addNode('event_extractor', agentNodes.event_extractor);
  workflow.addNode('quality_evaluator', agentNodes.quality_evaluator);
  workflow.addNode('geo_extractor', agentNodes.geo_extractor);
  
  // 认知体层
  workflow.addNode('timeline_analyst', agentNodes.timeline_analyst);
  workflow.addNode('causal_analyst', agentNodes.causal_analyst);
  workflow.addNode('knowledge_extractor', agentNodes.knowledge_extractor);
  workflow.addNode('result_validator', agentNodes.result_validator);
  
  // 决策体层
  workflow.addNode('scenario_simulator', agentNodes.scenario_simulator);
  workflow.addNode('sensitivity_analyst', agentNodes.sensitivity_analyst);
  workflow.addNode('action_advisor', agentNodes.action_advisor);
  
  // 行动体层
  workflow.addNode('report_generator', agentNodes.report_generator);
  workflow.addNode('document_executor', agentNodes.document_executor);
  workflow.addNode('quality_controller', agentNodes.quality_controller);
  workflow.addNode('execution_monitor', agentNodes.execution_monitor);
  
  // 进化体层
  workflow.addNode('knowledge_manager', agentNodes.knowledge_manager);
  workflow.addNode('review_analyst', agentNodes.review_analyst);
  
  // ==================== 设置入口 ====================
  
  workflow.setEntryPoint('intent_parser');
  
  // ==================== 添加边 ====================
  
  // 调度层 -> 感知体层
  workflow.addConditionalEdges(
    'intent_parser',
    (state: WorkflowState) => {
      const schedule = state.agentSchedule;
      if (!schedule) return 'scout_cluster';
      
      // 根据调度结果选择第一个感知体
      const perceptionAgents = ['scout_cluster', 'event_extractor', 'quality_evaluator', 'geo_extractor'];
      for (const agent of schedule.requiredAgents) {
        if (perceptionAgents.includes(agent)) {
          return agent;
        }
      }
      return 'scout_cluster';
    },
    {
      scout_cluster: 'scout_cluster',
      event_extractor: 'event_extractor',
      quality_evaluator: 'quality_evaluator',
      geo_extractor: 'geo_extractor'
    }
  );
  
  // 感知体层内部流程
  workflow.addEdge('scout_cluster', 'event_extractor');
  workflow.addEdge('event_extractor', 'quality_evaluator');
  workflow.addEdge('quality_evaluator', 'geo_extractor');
  
  // 感知体层 -> 认知体层
  workflow.addConditionalEdges(
    'geo_extractor',
    (state: WorkflowState) => {
      if (shouldSkip(state)) return 'report_generator';
      return 'timeline_analyst';
    },
    {
      timeline_analyst: 'timeline_analyst',
      report_generator: 'report_generator'
    }
  );
  
  // 认知体层内部流程
  workflow.addEdge('timeline_analyst', 'knowledge_extractor');
  workflow.addEdge('knowledge_extractor', 'causal_analyst');
  workflow.addEdge('causal_analyst', 'result_validator');
  
  // 认知体层 -> 决策体层
  workflow.addConditionalEdges(
    'result_validator',
    (state: WorkflowState) => {
      if (shouldSkip(state)) return 'report_generator';
      return 'scenario_simulator';
    },
    {
      scenario_simulator: 'scenario_simulator',
      report_generator: 'report_generator'
    }
  );
  
  // 决策体层内部流程
  workflow.addEdge('scenario_simulator', 'sensitivity_analyst');
  workflow.addEdge('sensitivity_analyst', 'action_advisor');
  
  // 决策体层 -> 行动体层
  workflow.addEdge('action_advisor', 'report_generator');
  
  // 行动体层内部流程
  workflow.addEdge('report_generator', 'quality_controller');
  workflow.addEdge('quality_controller', 'document_executor');
  workflow.addEdge('document_executor', 'execution_monitor');
  
  // 行动体层 -> 进化体层
  workflow.addEdge('execution_monitor', 'knowledge_manager');
  workflow.addEdge('knowledge_manager', 'review_analyst');
  
  // 进化体层 -> 结束
  workflow.addEdge('review_analyst', END);
  
  // 编译图
  return workflow.compile();
}

  /**
   * 生成文本嵌入向量
   */
  async generateEmbedding(text: string): Promise<number[]> {
    try {
      const response = await fetch('https://ark.cn-beijing.volces.com/api/v3/embeddings', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${process.env.DOUBAO_API_KEY}`
        },
        body: JSON.stringify({

          input: text.slice(0, 8000), // 限制长度
          encoding_format: 'float'
        })
      });
      
      const data = await response.json();
      
      if (data.data && data.data[0] && data.data[0].embedding) {
        return data.data[0].embedding;
      }
      
      throw new Error('Invalid embedding response');
      
    } catch (error) {
      console.error('[EmbeddingService] Generate embedding error:', error);
      throw error;
    }
  }
  
  /**
   * 批量生成嵌入向量
   */
  async generateEmbeddings(texts: string[]): Promise<number[][]> {
    const embeddings: number[][] = [];
    
    // 批量处理，每次最多5个
    for (let i = 0; i < texts.length; i += 5) {
      const batch = texts.slice(i, i + 5);
      const batchEmbeddings = await Promise.all(
        batch.map(text => this.generateEmbedding(text))
      );
      embeddings.push(...batchEmbeddings);
    }
    
    return embeddings;
  }
  
  /**
   * 计算余弦相似度
   */
  cosineSimilarity(a: number[], b: number[]): number {
    if (a.length !== b.length) {
      throw new Error('Vector dimensions must match');
    }
    
    let dotProduct = 0;
    let normA = 0;
    let normB = 0;
    
    for (let i = 0; i < a.length; i++) {
      dotProduct += a[i] * b[i];
      normA += a[i] * a[i];
      normB += b[i] * b[i];
    }
    
    return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));
  }

  async indexDocument(

  /**
   * 检测实体冲突
   */
  private async detectEntityConflicts(
    knowledge: ExtractedKnowledge,
    sourceCaseId: string,
    supabase: any
  ): Promise<KnowledgeConflict[]> {
    const conflicts: KnowledgeConflict[] = [];
    
    // 搜索相似实体
    const { data: existingEntities, error } = await supabase
      .from('kg_entities')
      .select('id, name, description, type, created_at')
      .ilike('name', `%${knowledge.title}%`)
      .limit(5);
    
    if (error || !existingEntities) return conflicts;
    
    for (const entity of existingEntities) {
      // 检查是否有实质冲突（描述矛盾）
      const hasConflict = await this.checkFactConflict(
        entity.description || '',
        knowledge.content,
        knowledge.title
      );
      
      if (hasConflict) {
        conflicts.push({
          id: `conflict_${Date.now()}_${Math.random().toString(36).substr(2, 6)}`,
          type: 'entity_conflict',
          existingKnowledge: {
            id: entity.id,
            content: entity.description || entity.name,
            source: '知识图谱',
            createdAt: entity.created_at,
          },
          newKnowledge: {
            content: knowledge.content,
            source: 'analysis_extraction',
            sourceCaseId,
          },

          severity: 'medium',
          status: 'pending',
          createdAt: new Date().toISOString(),
        });
      }
    }
    
    return conflicts;
  }
  
  /**
   * 检测关系冲突
   */
  private async detectRelationConflicts(
    knowledge: ExtractedKnowledge,
    sourceCaseId: string,
    supabase: any
  ): Promise<KnowledgeConflict[]> {
    const conflicts: KnowledgeConflict[] = [];
    
    // 解析关系中的实体
    const relationMatch = knowledge.title.match(/(.+?)\s*[→\->]+\s*(.+)/);
    if (!relationMatch) return conflicts;
    
    const [, sourceEntity, targetEntity] = relationMatch;
    
    // 搜索相反方向的关系
    const { data: reverseRelations, error } = await supabase
      .from('kg_relations')
      .select(`
        id, type, confidence, evidence, created_at,
        source_entity:kg_entities!kg_relations_source_entity_id_fkey(name),
        target_entity:kg_entities!kg_relations_target_entity_id_fkey(name)
      `)
      .ilike('kg_entities.name', `%${targetEntity.trim()}%`);
    
    if (error || !reverseRelations) return conflicts;
    
    for (const relation of reverseRelations) {
      const sourceName = relation.source_entity?.name || '';
      if (sourceName.toLowerCase().includes(targetEntity.trim().toLowerCase())) {
        conflicts.push({
          id: `conflict_${Date.now()}_${Math.random().toString(36).substr(2, 6)}`,
          type: 'relation_conflict',
          existingKnowledge: {
            id: relation.id,
            content: `${sourceName} → ${relation.target_entity?.name}`,
            source: '知识图谱',
            createdAt: relation.created_at,
          },
          newKnowledge: {
            content: knowledge.content,
            source: 'analysis_extraction',
            sourceCaseId,
          },
          conflictDescription: `发现相反方向的关系`,
          severity: 'medium',
          status: 'pending',
          createdAt: new Date().toISOString(),
        });
      }
    }
    
    return conflicts;
  }
  
  /**
   * 检查事实冲突（使用LLM）
   */
  private async checkFactConflict(
    existingContent: string,
    newContent: string,
    topic: string
  ): Promise<boolean> {
    if (!existingContent || !newContent) return false;

${existingContent}

描述2（新知识）：
${newContent}

如果存在矛盾（如数据不同、观点相反、结论冲突），请回答"是冲突"。
如果是互补或一致的信息，请回答"不冲突"。
只需回答"是冲突"或"不冲突"。`;
    
    try {
      const response = await llmClient.invoke([{

        content: prompt
      }], {
        model: 'doubao-seed-2-0-pro-260215',
        temperature: 0.1
      });
      
      const text = response.content || '';
      return text.includes('是冲突');
      
    } catch (error) {
      console.error('[KnowledgeAccumulation] Check conflict error:', error);
      return false;
    }
  }
  
  // ==================== 专家确认 ====================
  
  /**

  async createExpertConfirmationRequest(

          newKnowledge: conflict.newKnowledge,
        },

      });
      
    } catch (error) {
      console.error('[KnowledgeAccumulation] Create confirmation request error:', error);
    }
  }
  
  // ==================== 知识存储 ====================
  
  /**
   * 存储知识到知识图谱
   */
  async storeKnowledge(
    knowledge: ExtractedKnowledge[],
    sourceCaseId: string
  ): Promise<void> {
    const supabase = this.getSupabase();
    if (!supabase) return;
    
    for (const k of knowledge) {
      try {
        if (k.type === 'entity') {
          // 存储实体
          await supabase.from('kg_entities').upsert({
            name: k.title,
            type: '其他',
            description: k.content,
            importance: k.metadata.confidence.toString(),
            source_type: 'llm',
            verified: false,
          }, { onConflict: 'name' });
          
        } else if (k.type === 'relationship') {
          // 解析并存储关系
          const relationMatch = k.title.match(/(.+?)\s*[→\->]+\s*(.+)/);
          if (relationMatch) {
            const [, source, target] = relationMatch;
            
            // 创建或获取源实体
            const { data: sourceEntity } = await supabase
              .from('kg_entities')
              .upsert({ name: source.trim(), type: '其他', source_type: 'llm' }, { onConflict: 'name' })
              .select('id')
              .maybeSingle();
            
            // 创建或获取目标实体
            const { data: targetEntity } = await supabase
              .from('kg_entities')
              .upsert({ name: target.trim(), type: '其他', source_type: 'llm' }, { onConflict: 'name' })
              .select('id')
              .maybeSingle();
            
            if (sourceEntity && targetEntity) {
              // 创建关系
              await supabase.from('kg_relations').upsert({
                source_entity_id: sourceEntity.id,
                target_entity_id: targetEntity.id,
                type: '影响',
                confidence: k.metadata.confidence.toString(),
                evidence: k.content,
                source_type: 'llm',
              });
            }
          }
        }
      } catch (error) {
        console.error('[KnowledgeAccumulation] Store knowledge error:', error);
      }
    }
  }
}

/**

 * 基于Judea Pearl的结构因果模型(SCM)实现因果推断

import { createLLMClient, LLMClient } from '@/lib/llm';
import { getSupabaseClient } from '@/storage/database/supabase-client';

// ==================== 类型定义 ====================

export interface CausalVariable {
  name: string;
  type: 'continuous' | 'binary' | 'categorical';
  description: string;
  possibleValues?: string[];
  observed?: boolean;
}

export interface CausalRelation {
  cause: string;
  effect: string;
  mechanism: string;
  strength: number;
  confidence: number;
  evidence?: string[];
}

export interface CausalGraph {
  variables: CausalVariable[];
  edges: CausalRelation[];
  latentVariables?: string[];
}

export interface Intervention {
  variable: string;
  value: string | number;
  description: string;
}

export interface CounterfactualResult {
  premise: string;
  actualOutcome: string;
  counterfactualOutcome: string;
  probability: number;
  explanation: string;
}

export interface InferenceResult {
  query: string;
  answer: string;
  confidence: number;
  reasoning: string[];
  evidence: string[];
}

// ==================== 因果推断引擎 ====================

/**
 * 结构因果模型推断引擎
 */
export class CausalInferenceEngine {
  private llmClient: LLMClient | null = null;
  private graph: CausalGraph | null = null;
  
  /**
   * 获取LLM客户端
   */
  private getLLMClient(): LLMClient {
    if (!this.llmClient) {
      this.llmClient = createLLMClient({});
    }
    return this.llmClient;
  }
  
  /**
   * 设置因果图
   */
  setGraph(graph: CausalGraph): void {
    this.graph = graph;
  }
  
  /**
   * 从分析结果构建因果图
   */
  async buildCausalGraph(
    factors: Array<{ factor: string; description: string; impact: string }>,
    events: Array<{ event: string; description?: string }>
  ): Promise<CausalGraph> {
    const llmClient = this.getLLMClient();
    
    const prompt = `基于以下关键因素和事件，构建因果图。

关键因素：
${factors.map(f => `- ${f.factor} (${f.impact}): ${f.description}`).join('\n')}

事件：
${events.map(e => `- ${e.event}: ${e.description || ''}`).join('\n')}

请输出JSON格式的因果图：
{
  "variables": [
    {
      "name": "变量名",
      "type": "continuous|binary|categorical",
      "description": "变量描述"
    }
  ],
  "edges": [
    {
      "cause": "原因变量",
      "effect": "结果变量",
      "mechanism": "因果机制描述",
      "strength": 0.8,
      "confidence": 0.7
    }
  ]
}`;
    
    try {
      const response = await llmClient.invoke([{
        role: 'user',
        content: prompt
      }], {
        model: 'doubao-seed-2-0-pro-260215',
        temperature: 0.2
      });
      
      const text = response.content || '';
      const jsonMatch = text.match(/\{[\s\S]*\}/);
      
      if (jsonMatch) {
        const graph = JSON.parse(jsonMatch[0]);
        this.graph = graph;
        return graph;
      }
      
      // 返回默认图
      return this.createDefaultGraph(factors);
      
    } catch (error) {
      console.error('[CausalInference] Build graph error:', error);
      return this.createDefaultGraph(factors);
    }
  }
  
  /**
   * 创建默认因果图
   */
  private createDefaultGraph(
    factors: Array<{ factor: string; description: string }>
  ): CausalGraph {
    const variables: CausalVariable[] = factors.map(f => ({
      name: f.factor,
      type: 'continuous' as const,
      description: f.description
    }));
    
    const edges: CausalRelation[] = [];
    for (let i = 0; i < factors.length - 1; i++) {
      edges.push({
        cause: factors[i].factor,
        effect: factors[i + 1].factor,
        mechanism: '影响',
        strength: 0.5,
        confidence: 0.5
      });
    }
    
    return { variables, edges };
  }
  
  // ==================== 干预分析 ====================
  
  /**
   * 执行干预分析 (do-calculus)
   */
  async doIntervention(
    intervention: Intervention,
    targetVariable: string
  ): Promise<InferenceResult> {
    if (!this.graph) {
      return {
        query: `do(${intervention.variable}=${intervention.value})`,
        answer: '无法执行干预分析：因果图未初始化',
        confidence: 0,
        reasoning: [],
        evidence: []
      };
    }
    
    const llmClient = this.getLLMClient();
    
    // 构建干预后的因果图描述
    const graphDesc = this.describeGraph();
    
    const prompt = `执行因果干预分析。

因果图：
${graphDesc}

干预操作：do(${intervention.variable} = ${intervention.value})
干预描述：${intervention.description}
目标变量：${targetVariable}

请分析：
1. 干预对目标变量的影响
2. 影响的传导路径
3. 置信度评估

输出JSON格式：
{
  "answer": "干预结果描述",
  "confidence": 0.75,
  "reasoning": ["推理步骤1", "推理步骤2"],
  "evidence": ["证据1", "证据2"]
}`;
    
    try {
      const response = await llmClient.invoke([{
        role: 'user',
        content: prompt
      }], {
        model: 'doubao-seed-2-0-pro-260215',
        temperature: 0.2
      });
      
      const text = response.content || '';
      const jsonMatch = text.match(/\{[\s\S]*\}/);
      
      if (jsonMatch) {
        const result = JSON.parse(jsonMatch[0]);
        return {
          query: `do(${intervention.variable}=${intervention.value})`,
          ...result
        };
      }
      
      return {
        query: `do(${intervention.variable}=${intervention.value})`,
        answer: '干预分析失败',
        confidence: 0,
        reasoning: [],
        evidence: []
      };
      
    } catch (error) {
      console.error('[CausalInference] Intervention error:', error);
      return {
        query: `do(${intervention.variable}=${intervention.value})`,
        answer: `干预分析错误: ${error}`,
        confidence: 0,
        reasoning: [],
        evidence: []
      };
    }
  }
  
  // ==================== 反事实推理 ====================
  
  /**
   * 执行反事实推理
   */
  async counterfactual(
    observedFacts: Array<{ variable: string; value: string }>,
    hypotheticalIntervention: Intervention,
    targetVariable: string
  ): Promise<CounterfactualResult> {
    const llmClient = this.getLLMClient();
    
    const prompt = `执行反事实推理。

观察到的事实：
${observedFacts.map(f => `- ${f.variable} = ${f.value}`).join('\n')}

假设性干预：如果 ${hypotheticalIntervention.variable} 是 ${hypotheticalIntervention.value}
目标：${targetVariable} 会是什么？

请分析：
1. 实际世界中目标变量的值
2. 反事实世界中目标变量的值
3. 变化的概率和解释

输出JSON格式：
{
  "premise": "反事实前提",
  "actualOutcome": "实际结果",
  "counterfactualOutcome": "反事实结果",
  "probability": 0.7,
  "explanation": "解释说明"
}`;
    
    try {
      const response = await llmClient.invoke([{
        role: 'user',
        content: prompt
      }], {
        model: 'doubao-seed-2-0-pro-260215',
        temperature: 0.3
      });
      
      const text = response.content || '';
      const jsonMatch = text.match(/\{[\s\S]*\}/);
      
      if (jsonMatch) {
        return JSON.parse(jsonMatch[0]);
      }
      
      return {
        premise: hypotheticalIntervention.description,
        actualOutcome: '未知',
        counterfactualOutcome: '未知',
        probability: 0,
        explanation: '反事实推理失败'
      };
      
    } catch (error) {
      console.error('[CausalInference] Counterfactual error:', error);
      return {
        premise: hypotheticalIntervention.description,
        actualOutcome: '错误',
        counterfactualOutcome: '错误',
        probability: 0,
        explanation: String(error)
      };
    }
  }
  
  // ==================== 因果路径发现 ====================
  
  /**
   * 发现因果路径
   */
  async discoverCausalPath(
    startVariable: string,
    endVariable: string
  ): Promise<string[][]> {
    if (!this.graph) {
      return [];
    }
    
    const paths: string[][] = [];
    const visited = new Set<string>();
    
    // 深度优先搜索找路径
    const dfs = (current: string, path: string[]) => {
      if (current === endVariable) {
        paths.push([...path]);
        return;
      }
      
      visited.add(current);
      
      // 找到所有从当前变量出发的边
      const outgoingEdges = this.graph!.edges.filter(e => e.cause === current);
      
      for (const edge of outgoingEdges) {
        if (!visited.has(edge.effect)) {
          path.push(edge.effect);
          dfs(edge.effect, path);
          path.pop();
        }
      }
      
      visited.delete(current);
    };
    
    dfs(startVariable, [startVariable]);
    
    return paths;
  }
  
  /**
   * 计算直接和间接效应
   */
  async calculateEffects(
    cause: string,
    effect: string
  ): Promise<{
    directEffect: number;
    indirectEffect: number;
    totalEffect: number;
    pathDetails: Array<{ path: string[]; effect: number }>;
  }> {
    if (!this.graph) {
      return {
        directEffect: 0,
        indirectEffect: 0,
        totalEffect: 0,
        pathDetails: []
      };
    }
    
    // 查找直接效应
    const directEdge = this.graph.edges.find(
      e => e.cause === cause && e.effect === effect
    );
    const directEffect = directEdge?.strength || 0;
    
    // 查找间接路径
    const allPaths = await this.discoverCausalPath(cause, effect);
    const indirectPaths = allPaths.filter(p => p.length > 2);
    
    // 计算间接效应
    let indirectEffect = 0;
    const pathDetails: Array<{ path: string[]; effect: number }> = [];
    
    for (const path of indirectPaths) {
      // 路径效应 = 边强度的乘积
      let pathEffect = 1;
      for (let i = 0; i < path.length - 1; i++) {
        const edge = this.graph!.edges.find(
          e => e.cause === path[i] && e.effect === path[i + 1]
        );
        pathEffect *= edge?.strength || 0.5;
      }
      indirectEffect += pathEffect;
      pathDetails.push({ path, effect: pathEffect });
    }
    
    return {
      directEffect,
      indirectEffect,
      totalEffect: directEffect + indirectEffect,
      pathDetails
    };
  }
  
  // ==================== 工具方法 ====================
  
  /**
   * 描述因果图
   */
  private describeGraph(): string {
    if (!this.graph) return '无因果图';
    
    const varDesc = this.graph.variables
      .map(v => `- ${v.name}: ${v.description}`)
      .join('\n');
    
    const edgeDesc = this.graph.edges
      .map(e => `- ${e.cause} → ${e.effect} (强度: ${e.strength})`)
      .join('\n');
    
    return `变量：\n${varDesc}\n\n关系：\n${edgeDesc}`;
  }
  
  /**
   * 验证因果图有效性
   */
  validateGraph(): { valid: boolean; issues: string[] } {
    if (!this.graph) {
      return { valid: false, issues: ['因果图未初始化'] };
    }
    
    const issues: string[] = [];
    
    // 检查孤立节点
    const connectedVars = new Set<string>();
    for (const edge of this.graph.edges) {
      connectedVars.add(edge.cause);
      connectedVars.add(edge.effect);
    }
    
    for (const v of this.graph.variables) {
      if (!connectedVars.has(v.name)) {
        issues.push(`变量 ${v.name} 是孤立节点`);
      }
    }
    
    // 检查环
    const visited = new Set<string>();
    const recStack = new Set<string>();
    
    const hasCycle = (node: string): boolean => {
      visited.add(node);
      recStack.add(node);
      
      const outgoing = this.graph!.edges.filter(e => e.cause === node);
      for (const edge of outgoing) {
        if (!visited.has(edge.effect)) {
          if (hasCycle(edge.effect)) return true;
        } else if (recStack.has(edge.effect)) {
          return true;
        }
      }
      
      recStack.delete(node);
      return false;
    };
    
    for (const v of this.graph.variables) {
      if (!visited.has(v.name)) {
        if (hasCycle(v.name)) {
          issues.push('因果图存在环，不符合DAG要求');
          break;
        }
      }
    }
    
    return {
      valid: issues.length === 0,
      issues
    };
  }
}

/**
 * 格式化时间
 */
function formatTime(dateStr?: string): string {
  if (!dateStr) return '未知时间';
  
  try {
    const date = new Date(dateStr);
    const now = new Date();
    const diffMs = now.getTime() - date.getTime();
    const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24));
    const diffHours = Math.floor(diffMs / (1000 * 60 * 60));
    const diffMinutes = Math.floor(diffMs / (1000 * 60));
    
    if (diffMinutes < 1) return '刚刚';
    if (diffMinutes < 60) return `${diffMinutes}分钟前`;
    if (diffHours < 24) return `${diffHours}小时前`;
    if (diffDays < 7) return `${diffDays}天前`;
    if (diffDays < 30) return `${Math.floor(diffDays / 7)}周周前`;
    if (diffDays < 365) return `${Math.floor(diffDays / 30)}个月前`;
    return `${Math.floor(diffDays / 365)}年前`;
  } catch {
    return dateStr;
  }
}

/**
 * 获取结论摘要
 */
function getConclusionSummary(conclusion?: string | Record<string, unknown>): string {
  if (!conclusion) return '无结论';
  
  if (typeof conclusion === 'string') {
    return conclusion.slice(0, 150) + (conclusion.length > 150 ? '...' : '');
  }
  
  if (typeof conclusion === 'object') {
    const summary = (conclusion as Record<string, unknown>).summary;
    if (summary && typeof summary === 'string') {
      return summary.slice(0, 150) + (summary.length > 150 ? '...' : '');
    }
    return JSON.stringify(conclusion).slice(0, 150) + '...';
  }
  
  return '无结论';
}

/**
 * 获取相似度标签
 */
function getSimilarityBadge(similarity: number): { label: string; className: string } {
  if (similarity >= 0.9) {
    return { label: '高度相似', className: 'bg-green-500 text-white' };
  }
  if (similarity >= 0.75) {
    return { label: '较为相似', className: 'bg-blue-500 text-white' };
  }
  if (similarity >= 0.6) {
    return { label: '部分相似', className: 'bg-yellow-500 text-white' };
  }
  return { label: '略有相似', className: 'bg-gray-500 text-white' };
}
